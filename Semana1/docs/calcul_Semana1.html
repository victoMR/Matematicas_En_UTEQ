100    def extraer_coeficiente(self, term, operation):
101        """
102        Extrae el coeficiente de un término dado, considerando la operación
103        (+ o -) que lo precede. El coeficiente se ajusta según el signo de la operación.
104
105        :param term: El término que contiene el coeficiente (por ejemplo, "2x").
106        :param operation: La operación entre los términos, "+" o "-".
107        :return: El coeficiente como una fracción.
108        """
109        try:
110            coef = Fraction(term)
111            if operation == "-":
112                coef *= -1
113            return coef
114        except ValueError:
115            raise ValueError(f"Término inválido: {term}")

117    def mostrar_matriz(self, a1, b1, a2, b2, c1, c2):
118        """
119        Muestra la matriz de coeficientes y los valores de las ecuaciones en el área de resultados.
120
121        :param a1: Coeficiente de x en la primera ecuación.
122        :param b1: Coeficiente de y en la primera ecuación.
123        :param a2: Coeficiente de x en la segunda ecuación.
124        :param b2: Coeficiente de y en la segunda ecuación.
125        :param c1: Constante de la primera ecuación.
126        :param c2: Constante de la segunda ecuación.
127        """
128        # Separador visual
129        self.result_area.insert(tk.END, "---------------------\n")
130        # Mostrar las ecuaciones recibidas
131        self.result_area.insert(tk.END, f"Ecuaciones recibidas:\n")
132        self.result_area.insert(tk.END, f" {a1}x {b1}y = {c1}\n")
133        self.result_area.insert(tk.END, f" {a2}x {b2}y = {c2}\n")
134        # Separador visual
135        self.result_area.insert(tk.END, "---------------------\n")
136        # Mostrar la matriz Delta inicial
137        self.result_area.insert(tk.END, f"Matriz Delta inicial:\n")
138        self.result_area.insert(tk.END, f" | {a1}  {b1} |\n")
139        self.result_area.insert(tk.END, f" | {a2}  {b2} |\n")
140        # Separador visual
141        self.result_area.insert(tk.END, "---------------------\n")
142        # Mostrar la matriz de resultados
143        self.result_area.insert(tk.END, f"Igual a:\n | {c1} |\n | {c2} |\n")
144        # Separador visual
145        self.result_area.insert(tk.END, "---------------------\n")

147    def resolver_ecuaciones(self):
148        """
149        Resuelve el sistema de ecuaciones utilizando el método de matrices (regla de Cramer).
150        Calcula los determinantes necesarios para encontrar las soluciones de x e y,
151        y muestra el proceso y resultados en el área de resultados.
152        """
153        try:
154            # Extraer coeficientes y constantes para la primera ecuación
155            a1 = self.extraer_coeficiente(self.entry_a1.get(), "+")
156            b1 = self.extraer_coeficiente(self.entry_b1.get(), self.operacion1.get())
157            c1 = Fraction(self.entry_c1.get())
158
159            # Extraer coeficientes y constantes para la segunda ecuación
160            a2 = self.extraer_coeficiente(self.entry_a2.get(), "+")
161            b2 = self.extraer_coeficiente(self.entry_b2.get(), self.operacion2.get())
162            c2 = Fraction(self.entry_c2.get())
163        except ValueError as e:
164            # Mostrar error si hay problemas con la entrada
165            messagebox.showerror("Error", str(e))
166            return
167
168        # Limpiar el área de resultados antes de mostrar el nuevo proceso
169        self.result_area.delete(1.0, tk.END)
170        # Mostrar la matriz inicial
171        self.mostrar_matriz(a1, b1, a2, b2, c1, c2)
172
173        # Calcular el determinante de Delta
174        delta = a1 * b2 - a2 * b1
175        self.result_area.insert(
176            tk.END, f"\nDeterminante de Delta = {a1} * {b2} - {a2} * {b1} = {delta}\n"
177        )
178        if delta == 0:
179            self.result_area.insert(
180                tk.END,
181                "\nEl sistema no tiene solución (Delta = 0).\n",
182            )
183            return
184
185        # Calcular el determinante de Delta_x (intercambiando la columna de c)
186        delta_x1 = c1 * b2 - c2 * b1
187        self.result_area.insert(
188            tk.END, f"\nMatriz Delta_x (reemplazo de columna de x):\n"
189        )
190        self.result_area.insert(tk.END, f" | {c1}  {b1} |\n")
191        self.result_area.insert(tk.END, f" | {c2}  {b2} |\n")
192        self.result_area.insert(
193            tk.END,
194            f"\nDeterminante de Delta_x = {c1} * {b2} - {c2} * {b1} = {delta_x1}\n",
195        )
196
197        # Calcular el determinante de Delta_y (intercambiando la columna de y)
198        delta_y1 = a1 * c2 - a2 * c1
199        self.result_area.insert(
200            tk.END, f"\nMatriz Delta_y (reemplazo de columna de y):\n"
201        )
202        self.result_area.insert(tk.END, f" | {a1}  {c1} |\n")
203        self.result_area.insert(tk.END, f" | {a2}  {c2} |\n")
204        self.result_area.insert(
205            tk.END,
206            f"\nDeterminante de Delta_y = {a1} * {c2} - {a2} * {c1} = {delta_y1}\n",
207        )
208
209        # Calcular las soluciones finales
210        x = Fraction(delta_x1, delta)
211        y = Fraction(delta_y1, delta)
212
213        # Mostrar las soluciones finales
214        if x is not None and y is not None:
215            # Separador visual
216            self.result_area.insert(tk.END, "---------------------\n")
217            # Mostrar la solución para x
218            self.result_area.insert(
219                tk.END, f"\nSolución para x: x = Delta_x / Delta = {x}\n"
220            )
221            # Mostrar la solución para y
222            self.result_area.insert(
223                tk.END, f"Solución para y: y = Delta_y / Delta = {y}\n"
224            )
225        else:
226            self.result_area.insert(
227                tk.END,
228                f"\nEl sistema no tiene solución o tiene infinitas soluciones (Delta = 0).\n",
229            )
230            return
